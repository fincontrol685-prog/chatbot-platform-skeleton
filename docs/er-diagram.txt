Entidades principais (resumo ER - texto)

- USER_ACCOUNT (1) <--> (N) USER_ROLE (N) <--> (1) ROLE
- BOT (1) <--> (N) FLOW
- FLOW (1) <--> (N) FLOW_STEP
- INTENT (1) <--> (N) UTTERANCE
- BOT (1) <--> (N) INTENT
- SESSION (1) <--> (N) MESSAGE_HISTORY
- METRIC (pooled) relacionado a BOT/SESSION

Atributos críticos

- USER_ACCOUNT: id, username, email, password_hash, enabled, created_at, last_login
- ROLE: id, name (ADMIN, GESTOR, USUARIO), description
- BOT: id, name, key, enabled, owner_id, config (JSON optional), created_at
- FLOW: id, bot_id, name, type (tree|state-machine|script), definition (JSON/BLOB), enabled
- FLOW_STEP: id, flow_id, step_type (message, decision, action), payload (JSON), order
- INTENT: id, bot_id, name, confidence_threshold, created_at
- UTTERANCE: id, intent_id, text
- SESSION: id, bot_id, user_id, state (JSON), started_at, ended_at
- MESSAGE_HISTORY: id, session_id, sender (USER/BOT), message_text, metadata (JSON), timestamp
- METRIC: id, bot_id, metric_type, metric_value, captured_at

Oracle DDL inicial (simplificado)

-- V1: criar tabelas principais
CREATE TABLE ROLE (
    ID NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    NAME VARCHAR2(50) NOT NULL UNIQUE,
    DESCRIPTION VARCHAR2(4000),
    CREATED_AT TIMESTAMP DEFAULT SYSTIMESTAMP
);

CREATE TABLE USER_ACCOUNT (
    ID NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    USERNAME VARCHAR2(100) NOT NULL UNIQUE,
    EMAIL VARCHAR2(255),
    PASSWORD_HASH VARCHAR2(4000) NOT NULL,
    ENABLED NUMBER(1) DEFAULT 1,
    CREATED_AT TIMESTAMP DEFAULT SYSTIMESTAMP,
    LAST_LOGIN TIMESTAMP,
    VERSION NUMBER DEFAULT 0
);

CREATE TABLE USER_ROLE (
    USER_ID NUMBER NOT NULL,
    ROLE_ID NUMBER NOT NULL,
    CONSTRAINT PK_USER_ROLE PRIMARY KEY (USER_ID, ROLE_ID),
    CONSTRAINT FK_UR_USER FOREIGN KEY (USER_ID) REFERENCES USER_ACCOUNT(ID),
    CONSTRAINT FK_UR_ROLE FOREIGN KEY (ROLE_ID) REFERENCES ROLE(ID)
);

CREATE TABLE BOT (
    ID NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    NAME VARCHAR2(255) NOT NULL,
    BOT_KEY VARCHAR2(255) NOT NULL UNIQUE,
    ENABLED NUMBER(1) DEFAULT 1,
    OWNER_ID NUMBER,
    CONFIG CLOB,
    CREATED_AT TIMESTAMP DEFAULT SYSTIMESTAMP,
    VERSION NUMBER DEFAULT 0,
    CONSTRAINT FK_BOT_OWNER FOREIGN KEY (OWNER_ID) REFERENCES USER_ACCOUNT(ID)
);

CREATE TABLE FLOW (
    ID NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    BOT_ID NUMBER NOT NULL,
    NAME VARCHAR2(255) NOT NULL,
    TYPE VARCHAR2(50),
    DEFINITION CLOB,
    ENABLED NUMBER(1) DEFAULT 1,
    CREATED_AT TIMESTAMP DEFAULT SYSTIMESTAMP,
    VERSION NUMBER DEFAULT 0,
    CONSTRAINT FK_FLOW_BOT FOREIGN KEY (BOT_ID) REFERENCES BOT(ID)
);

CREATE TABLE INTENT (
    ID NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    BOT_ID NUMBER NOT NULL,
    NAME VARCHAR2(255) NOT NULL,
    CONFIDENCE_THRESHOLD NUMBER(5,4) DEFAULT 0.50,
    CREATED_AT TIMESTAMP DEFAULT SYSTIMESTAMP,
    CONSTRAINT FK_INTENT_BOT FOREIGN KEY (BOT_ID) REFERENCES BOT(ID)
);

CREATE TABLE UTTERANCE (
    ID NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    INTENT_ID NUMBER NOT NULL,
    TEXT VARCHAR2(4000) NOT NULL,
    CONSTRAINT FK_UTTER_INTENT FOREIGN KEY (INTENT_ID) REFERENCES INTENT(ID)
);

CREATE TABLE SESSION (
    ID NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    BOT_ID NUMBER NOT NULL,
    USER_ID NUMBER,
    STATE CLOB,
    STARTED_AT TIMESTAMP DEFAULT SYSTIMESTAMP,
    ENDED_AT TIMESTAMP,
    CONSTRAINT FK_SESSION_BOT FOREIGN KEY (BOT_ID) REFERENCES BOT(ID),
    CONSTRAINT FK_SESSION_USER FOREIGN KEY (USER_ID) REFERENCES USER_ACCOUNT(ID)
);

CREATE TABLE MESSAGE_HISTORY (
    ID NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    SESSION_ID NUMBER NOT NULL,
    SENDER VARCHAR2(50),
    MESSAGE_TEXT CLOB,
    METADATA CLOB,
    TIMESTAMP TIMESTAMP DEFAULT SYSTIMESTAMP,
    CONSTRAINT FK_MSG_SESSION FOREIGN KEY (SESSION_ID) REFERENCES SESSION(ID)
);

CREATE TABLE METRIC (
    ID NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    BOT_ID NUMBER,
    METRIC_TYPE VARCHAR2(100),
    METRIC_VALUE NUMBER,
    CAPTURED_AT TIMESTAMP DEFAULT SYSTIMESTAMP
);

-- Indexes
CREATE INDEX IDX_USER_USERNAME ON USER_ACCOUNT(USERNAME);
CREATE INDEX IDX_BOT_OWNER ON BOT(OWNER_ID);
CREATE INDEX IDX_SESSION_BOT ON SESSION(BOT_ID);
